# Configuração do Vercel para NEXTAUTH_URL dinâmico

# O Vercel automaticamente define NEXTAUTH_URL baseado no domínio do deployment
# Para preview deployments, não precisamos definir manualmente

# Para produção, usar o domínio principal
NEXTAUTH_URL=https://magiclawyer.vercel.app

# Para desenvolvimento local
# NEXTAUTH_URL=http://localhost:3000

# Secret para assinatura de tokens (deve ser o mesmo em todos os ambientes)
NEXTAUTH_SECRET=your-secret-key-here
